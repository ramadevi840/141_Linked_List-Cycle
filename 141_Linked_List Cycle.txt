cl/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public boolean hasCycle(ListNode head) {
        // TC-O(n)
        // when empty head
        if(head==null){
            return false;
        }
        ListNode slow = head;
        ListNode fast = head;
        // if loop contains then while loop run till fast and slow reaches same place 
        while(fast!=null && fast.next!=null){
            slow=slow.next;
            fast=fast.next.next;
            // fast and slow reach we return 
            if(slow==fast){
                return true;
            }
        }
        return false;
    }
}
